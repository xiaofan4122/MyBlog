一、顺序存储结构的线性表​

题干：​

给定一批数据如下：

学号姓名分数
101张小三95.88
102王小四96.99
103李小五95.12
104周兴国98.13
105王爱家93.56
106赵公明97.69

基于顺序存储结构的线性表之创建、元素插入及元素删除操作实验，内容包括：

（1）基于顺序存储结构创建一个最多可容纳1000人信息的线性表，将上述数据存入线性表中，并能够输出表中各个元素的值及该线性表的长度

（2）在上述线性表中第i个位置插入如下同学的信息，并输出该线性表的长度

学号姓名分数
196我爱国99.86

（3）删除上述线性表中第i个位置的信息，并输出该线性表的长度

问题一代码及结果：​

要求：基于顺序存储结构创建一个最多可容纳1000人信息的线性表，将上述数据存入线性表中，并能够输出表中各个元素的值及该线性表的长度

代码块​

对于顺序存储结构，其数据都是连续存储的，所以可以通过类似于数组的形式来建立线性表

头文件和宏定义

#include <stdio.h>
#include<stdlib.h>
#define LIST_SIZE 1000  //线性表存储空间的初始分配量

首先定义一个结构体类型，然后把已有数据存储下来

//定义学生
typedef struct{
    int number;
    char name[10];
    double score;
}Students;
//给定的数据
Students Data_List[6]={
        {101,"张小三",95.88},
        {102,"王小四",96.99},
        {103,"李小五",95.12},
        {104,"周兴国",98.13},
        {105,"王爱家",93.56},
        {106,"赵公明",97.69}
};

之后创建另一个结构体用作顺序存储结构线性表

typedef struct
{
    Students *elem;  //数组存储数据元素，最大值位MAXSIZE
    int length;              //线性表的长度
}SqList;                     //建立线性表结构体

函数返回状态均为Status，该值是一个自定义的枚举类型，只有OK和ERROR两个枚举量

typedef enum {   //Status是函数结果状态代码
    OK = 1,
    ERROR = 0
}Status;

表初始化函数

Status InitList(SqList *L) {
    L -> elem = (Students *)malloc(LIST_SIZE * sizeof(Students));
    if(!L -> elem){
        return ERROR;
    }
    L -> length = 0;
    return OK;
}

插入函数

Status ListInsert(SqList *L, int i, Students e) {
    int k;
    if (L->length == LIST_SIZE){  //线性表已满
        return ERROR;
    }
    if (i < 1 || i > L->length+1){ //当i不在范围内时
        return ERROR;
    }
    if (i <= L->length){  //若插入位置不在表尾
        for(k = L->length-1;k >= i-1;k--){
            L->elem[k+1] = L->elem[k];
        }
    }
    L->elem[i-1] = e;   //将新元素插入
    L->length++;    //长度加1
    return OK;
}

输出函数

void OutPut(SqList L) {
    printf("当前顺序表的长度:%d\n", L.length);
    for(int i = 0; i < L.length; i++){
        printf("学号：%d\t姓名：%s\t分数：%.2f\n",L.elem[i].number,L.elem[i].name,L.elem[i].score);
    }
    printf("\n");
}

主函数

int main(){
    SqList L;
    printf("初始化线性表\n");
    InitList(&L);
    for(int i=0;i<6;i++){
        ListInsert(&L,1,Data_List[i]);
    }
    printf("插入已有的六个学生\n");
    OutPut(L);
    getchar();
    return 0;
}

基于链式存储结构的线性表之创建操作实验，内容包括：

（1）针对上述给出的数据，基于链式存储结构创建一个线性表，将上述数据存入线性表中，并能够输出链表各节点所含数据的值。

（2）设计一个函数能够输出上述链表的长度。

（3）在上述链表第一个节点之前插入包含如下数据的节点，并输出该线性表的长度：

学号姓名分数
196我爱国99.86

问题一代码及结果​

要求：针对上述给出的数据，基于链式存储结构创建一个线性表，将上述数据存入线性表中，并能够输出链表各节点所含数据的值。

代码块​

定义节点和相应的指针

typedef struct MY_Node * P_MY_Node;
struct MY_Node{
    Students student;
    P_MY_Node Next;
};

typedef P_MY_Node Position;//这里的位置是结点的地址 
typedef P_MY_Node List;

初始化代码

//初始化
List MakeEmpty(){
    List L;
    L = (List)malloc(sizeof (struct MY_Node));
    if (!L)
        exit (-1);
    L->Next = NULL;
    return L;
}

插入代码

//插入
void Insert(List L ,Students X,int i){
    Position tmp,pre;
    int cnt =0 ;
    pre = L;
    while(pre&&cnt<i-1){
        pre = pre->Next;
        cnt++;
    }
    if(pre==NULL||cnt!=i-1){
        printf("插入位置参数错误\n");
    }
    else{
        tmp = (Position)malloc(sizeof(struct MY_Node));
        tmp->student=X;
        tmp->Next=pre->Next;
        pre->Next=tmp;
    }
}

查找代码

//根据指定的位序查找
void FindIndex(List L,int K){
    Position p;
    int cnt = 1;//位序从1开始 
    p = L->Next;
    while(p&&cnt<K){
        p = p-> Next;
        cnt++;
    }
    if((cnt==K)&&p) printf("查找成功：学号：%d\t姓名：%s\t分数：%.2f\n", L->student.number, L->student.name, L->student.score);
    else printf("您查找数不存在");
}

输出代码

void DisLinkList(List L)
{
    List p = L->Next;
    printf("输出链表： \n");
    while (p)
    {
        printf("学号：%d\t姓名：%s\t分数：%.2f\n", p->student.number, p->student.name, p->student.score);
        p = p->Next;
    }
}

主函数

int main(){
    printf("**********链式存储结构**********\n");
    Position L2 = MakeEmpty();
    int i;
    //插入
    for(i=0;i<6;i++){
        Insert(L2,Data_List[i],i+1);
    }
    //求表长
    printf("表长为：%d\n",Length(L2));
    //输出
    DisLinkList(L2);
    getchar();
    return 0;
}
